import requests # ========================================================= # CONFIG # ========================================================= BASE_URL = "http://your-bo-server:6405/biprws" USERNAME = "administrator" PASSWORD = "password" AUTH_TYPE = "secEnterprise" # ========================================================= # GLOBAL STRUCTURES # ========================================================= folder_tree = {} object_map = {} name_map = {} parent_map = {} visited_folders = set() # ========================================================= # LOGIN # ========================================================= def login(): url = f"{BASE_URL}/logon/long" payload = { "userName": USERNAME, "password": PASSWORD, "auth": AUTH_TYPE } headers = { "Content-Type": "application/json", "Accept": "application/json" } response = requests.post(url, json=payload, headers=headers) response.raise_for_status() token = response.json()["logonToken"] return { "X-SAP-LogonToken": token, "Accept": "application/json" } # ========================================================= # GET ALL FOLDERS # ========================================================= def get_all_folders(headers): url = f"{BASE_URL}/folders" response = requests.get(url, headers=headers) response.raise_for_status() return response.json().get("entries", []) # ========================================================= # GET CHILDREN # ========================================================= def get_children(headers, folder_id): url = f"{BASE_URL}/folders/{folder_id}/children" response = requests.get(url, headers=headers) response.raise_for_status() return response.json().get("entries", []) # ========================================================= # RECURSIVE TRAVERSAL # ========================================================= def traverse_folder(headers, folder): folder_id = folder.get("id") folder_cuid = folder.get("cuid") folder_name = folder.get("name") if not folder_cuid: return # Avoid loops / duplicate traversal if folder_cuid in visited_folders: return visited_folders.add(folder_cuid) # Initialize folder node if folder_cuid not in folder_tree: folder_tree[folder_cuid] = { "name": folder_name, "children_folders": [], "children_objects": [] } name_map[folder_cuid] = folder_name print(f"\n[FOLDER] {folder_name}") # ===================================================== # GET CHILDREN # ===================================================== children = get_children(headers, folder_id) for child in children: child_type = child.get("type", "").lower() child_name = child.get("name") child_cuid = child.get("cuid") if not child_cuid: continue name_map[child_cuid] = child_name parent_map[child_cuid] = folder_cuid # ================================================= # SUB FOLDER # ================================================= if child_type == "folder": folder_tree[folder_cuid]["children_folders"].append( child_cuid ) traverse_folder(headers, child) # ================================================= # OBJECT # ================================================= else: folder_tree[folder_cuid]["children_objects"].append( child_cuid ) object_map[child_cuid] = { "name": child_name, "type": child_type, "parent_folder": folder_cuid } print(f" └── [{child_type.upper()}] {child_name}") # ========================================================= # BUILD COMPLETE PATH # ========================================================= def get_complete_path(cuid): if cuid not in name_map: return None path_parts = [] current = cuid while current: path_parts.append(name_map[current]) current = parent_map.get(current) path_parts.reverse() return "/" + "/".join(path_parts) # ========================================================= # MAIN # ========================================================= def main(): headers = login() print("\nFetching all folders...\n") folders = get_all_folders(headers) print(f"Total folders received: {len(folders)}") # ===================================================== # TRAVERSE EACH ROOT FOLDER # ===================================================== for folder in folders: try: traverse_folder(headers, folder) except Exception as e: print(f"Error processing folder: {e}") print("\n===================================") print("INDEXING COMPLETE") print("===================================\n") print(f"Folders Indexed : {len(folder_tree)}") print(f"Objects Indexed : {len(object_map)}") # ===================================================== # SEARCH LOOP # ===================================================== while True: cuid = input("\nEnter CUID (or exit): ").strip() if cuid.lower() == "exit": break path = get_complete_path(cuid) if path: print(f"\nFULL PATH:\n{path}") else: print("CUID not found") # ========================================================= if __name__ == "__main__": main()
