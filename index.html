import csv import requests from pprint import pprint # ========================================================= # CONFIG # ========================================================= BASE_URL = "http://your-bo-server:6405" USERNAME = "administrator" PASSWORD = "password" AUTH_TYPE = "secEnterprise" OUTPUT_CSV = "bo_schedule_report.csv" # ========================================================= # STATUS MAP # ========================================================= STATUS_MAP = { 0: "Pending", 1: "Running", 2: "Success", 3: "Failed", 4: "Paused", 5: "Retry", 8: "Recurring", 9: "Scheduled" } # ========================================================= # CLIENT # ========================================================= class BusinessObjectsClient: def __init__(self): self.headers = self.login() self.folder_tree = {} self.folder_name_map = {} self.folder_parent_map = {} self.visited_folders = set() # ===================================================== # LOGIN # ===================================================== def login(self): url = f"{BASE_URL}/biprws/logon/long" payload = { "userName": USERNAME, "password": PASSWORD, "auth": AUTH_TYPE } headers = { "Content-Type": "application/json", "Accept": "application/json" } response = requests.post( url, json=payload, headers=headers ) response.raise_for_status() token = response.json()["logonToken"] return { "X-SAP-LogonToken": token, "Accept": "application/json" } # ===================================================== # BUILD FOLDER MAP # ===================================================== def build_folder_map(self): print("Building folder map...") url = f"{BASE_URL}/biprws/v1/folders" response = requests.get( url, headers=self.headers ) response.raise_for_status() folders = response.json().get( "entries", [] ) for folder in folders: self._traverse_folder(folder) print( f"Folder map complete. " f"Total folders indexed: " f"{len(self.folder_tree)}" ) # ===================================================== # RECURSIVE TRAVERSAL # ===================================================== def _traverse_folder(self, folder): folder_id = folder.get("id") folder_cuid = folder.get("cuid") folder_name = folder.get("name") if not folder_cuid: return if folder_cuid in self.visited_folders: return self.visited_folders.add(folder_cuid) self.folder_name_map[ folder_cuid ] = folder_name self.folder_tree[folder_cuid] = { "id": folder_id, "name": folder_name, "children_folders": [], "children_objects": [] } children = self._get_children(folder_id) for child in children: child_type = str( child.get("type", "") ).lower() child_cuid = child.get("cuid") child_name = child.get("name") if not child_cuid: continue self.folder_name_map[ child_cuid ] = child_name self.folder_parent_map[ child_cuid ] = folder_cuid # ============================================= # SUB FOLDER # ============================================= if child_type == "folder": self.folder_tree[ folder_cuid ][ "children_folders" ].append(child_cuid) self._traverse_folder(child) # ============================================= # OBJECT # ============================================= else: self.folder_tree[ folder_cuid ][ "children_objects" ].append(child_cuid) # ===================================================== # GET CHILDREN # ===================================================== def _get_children(self, folder_id): url = ( f"{BASE_URL}/biprws/v1/folders/" f"{folder_id}/children" ) response = requests.get( url, headers=self.headers ) response.raise_for_status() return response.json().get( "entries", [] ) # ===================================================== # GET PATH # ===================================================== def get_folder_path(self, cuid): if cuid not in self.folder_name_map: return None path_parts = [] current = cuid while current: name = self.folder_name_map.get( current ) if name: path_parts.append(name) current = self.folder_parent_map.get( current ) path_parts.reverse() return "/" + "/".join(path_parts) # ===================================================== # EXECUTE CMS QUERY # ===================================================== def execute_cms_query(self, query): print("Executing CMS Query...") url = f"{BASE_URL}/biprws/v1/cmsquery" payload = { "query": query } all_records = [] page = 1 pagesize = 1000 while True: params = { "page": page, "pagesize": pagesize } response = requests.post( url, json=payload, params=params, headers=self.headers ) response.raise_for_status() data = response.json() entries = data.get( "entries", [] ) if not entries: break all_records.extend(entries) print( f"Fetched Page {page} " f"Records: {len(entries)}" ) if len(entries) < pagesize: break page += 1 print( f"Total CMS Records: " f"{len(all_records)}" ) return all_records # ===================================================== # GET STATUS # ===================================================== def get_status_value(self, code): return STATUS_MAP.get( code, f"Unknown ({code})" ) # ===================================================== # GET EXPIRY # ===================================================== def get_expiry( self, parent_id, instance_id ): try: url = ( f"{BASE_URL}/biprws/v1/documents/" f"{parent_id}/schedules/" f"{instance_id}" ) response = requests.get( url, headers=self.headers ) response.raise_for_status() data = response.json() return data.get("expiry") except: return None # ===================================================== # GET RECURRENCE # ===================================================== def get_recurrence( self, parent_id, instance_id ): try: url = ( f"{BASE_URL}/biprws/raylight/v1/" f"documents/{parent_id}/" f"schedules/{instance_id}" ) response = requests.get( url, headers=self.headers ) response.raise_for_status() data = response.json() return ( data.get("recurrence") or data.get("description") or data.get("schedule") ) except: return None # ===================================================== # BUILD SCHEMA # ===================================================== def build_record_schema(self, record): status_code = record.get( "SI_UISTATUS" ) parent_id = record.get( "SI_PARENTID" ) instance_id = record.get( "SI_ID" ) recurrence = None if status_code == 9: recurrence = self.get_recurrence( parent_id, instance_id ) return { "id": record.get("SI_ID"), "cuid": record.get("SI_CUID"), "name": record.get("SI_NAME"), "type": record.get("SI_KIND"), "status_code": status_code, "status": self.get_status_value( status_code ), "location": self.get_folder_path( record.get("SI_CUID") ), "author": record.get("SI_AUTHOR"), "owner": record.get("SI_OWNER"), "creation_time": record.get( "SI_CREATION_TIME" ), "endtime": record.get( "SI_ENDTIME" ), "starttime": record.get( "SI_STARTTIME" ), "nextruntime": record.get( "SI_NEXTRUNTIME" ), "recurrence": recurrence, "expiry": self.get_expiry( parent_id, instance_id ), "server": ( record.get( "SI_SCHEDULEINFO", {} ).get( "SI_MACHINEUSED" ) ), "error": ( record.get( "SI_STATUSINFO", {} ).get( "SI_SUBST_STRINGS" ) ) } # ===================================================== # EXPORT CSV # ===================================================== def export_to_csv( self, records, output_file ): if not records: print("No records to export") return print( f"Exporting CSV: " f"{output_file}" ) headers = records[0].keys() with open( output_file, mode="w", newline="", encoding="utf-8" ) as csvfile: writer = csv.DictWriter( csvfile, fieldnames=headers ) writer.writeheader() for row in records: writer.writerow(row) print( f"CSV Export Complete: " f"{output_file}" ) # ========================================================= # CMS QUERY # ========================================================= CMS_QUERY = """ SELECT SI_ID, SI_CUID, SI_NAME, SI_KIND, SI_UISTATUS, SI_AUTHOR, SI_OWNER, SI_CREATION_TIME, SI_ENDTIME, SI_STARTTIME, SI_NEXTRUNTIME, SI_PARENTID, SI_SCHEDULEINFO.SI_MACHINEUSED, SI_STATUSINFO FROM CI_APPOBJECTS, CI_INFOOBJECTS, CI_SYSTEMOBJECTS WHERE SI_INSTANCE = 1 """ # ========================================================= # MAIN # ========================================================= if __name__ == "__main__": client = BusinessObjectsClient() # ===================================================== # STEP 1 # ===================================================== client.build_folder_map() # ===================================================== # STEP 2 # ===================================================== cms_records = client.execute_cms_query( CMS_QUERY ) # ===================================================== # STEP 3 # ===================================================== final_records = [] for record in cms_records: try: schema = client.build_record_schema( record ) final_records.append(schema) except Exception as e: print( f"Schema build failed: {e}" ) # ===================================================== # STEP 4 # ===================================================== client.export_to_csv( final_records, OUTPUT_CSV ) # ===================================================== # SAMPLE # ===================================================== if final_records: print("\nSample Record:\n") pprint(final_records[0])
