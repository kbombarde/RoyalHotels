import requests # ========================================================= # CONFIG # ========================================================= BASE_URL = "http://your-bo-server:6405/biprws" USERNAME = "administrator" PASSWORD = "password" AUTH_TYPE = "secEnterprise" ROOT_FOLDER_CUID = "YOUR_ROOT_CUID" # ========================================================= # GLOBAL STRUCTURES # ========================================================= folder_tree = {} object_map = {} name_map = {} parent_map = {} # ========================================================= # LOGIN # ========================================================= def login(): url = f"{BASE_URL}/logon/long" payload = { "userName": USERNAME, "password": PASSWORD, "auth": AUTH_TYPE } headers = { "Content-Type": "application/json", "Accept": "application/json" } response = requests.post(url, json=payload, headers=headers) response.raise_for_status() token = response.json()["logonToken"] return { "X-SAP-LogonToken": token, "Accept": "application/json" } # ========================================================= # GET ROOT FOLDER # ========================================================= def get_folder_by_cuid(headers, cuid): url = f"{BASE_URL}/folders" response = requests.get( url, headers=headers, params={"cuid": cuid} ) response.raise_for_status() entries = response.json().get("entries", []) if not entries: return None return entries[0] # ========================================================= # GET CHILDREN # ========================================================= def get_children(headers, folder_id): url = f"{BASE_URL}/folders/{folder_id}/children" response = requests.get(url, headers=headers) response.raise_for_status() return response.json().get("entries", []) # ========================================================= # RECURSIVE TRAVERSAL # ========================================================= def traverse_folder(headers, folder_id, folder_cuid, parent_cuid=None): children = get_children(headers, folder_id) folder_tree[folder_cuid] = { "name": name_map[folder_cuid], "parent": parent_cuid, "children_folders": [], "children_objects": [] } for child in children: child_type = child.get("type", "").lower() child_name = child.get("name") child_cuid = child.get("cuid") child_id = child.get("id") if not child_cuid: continue name_map[child_cuid] = child_name parent_map[child_cuid] = folder_cuid # ================================================= # FOLDER # ================================================= if child_type == "folder": folder_tree[folder_cuid]["children_folders"].append(child_cuid) print(f"[FOLDER] {child_name}") traverse_folder( headers, child_id, child_cuid, folder_cuid ) # ================================================= # OBJECT # ================================================= else: folder_tree[folder_cuid]["children_objects"].append(child_cuid) object_map[child_cuid] = { "name": child_name, "type": child_type, "parent_folder": folder_cuid } print(f" └── [{child_type.upper()}] {child_name}") # ========================================================= # BUILD COMPLETE PATH # ========================================================= def get_complete_path(cuid): if cuid not in name_map: return None path_parts = [] current = cuid while current: path_parts.append(name_map[current]) current = parent_map.get(current) path_parts.reverse() return "/" + "/".join(path_parts) # ========================================================= # MAIN # ========================================================= def main(): headers = login() root = get_folder_by_cuid(headers, ROOT_FOLDER_CUID) if not root: print("Root folder not found") return root_id = root["id"] root_cuid = root["cuid"] root_name = root["name"] name_map[root_cuid] = root_name print(f"\nStarting traversal from: {root_name}\n") traverse_folder( headers, root_id, root_cuid ) print("\n===================================") print("Traversal Complete") print("===================================\n") print(f"Folders Indexed : {len(folder_tree)}") print(f"Objects Indexed : {len(object_map)}") while True: cuid = input("\nEnter CUID (or exit): ").strip() if cuid.lower() == "exit": break path = get_complete_path(cuid) if path: print(f"\nPATH:\n{path}") else: print("CUID not found") # ========================================================= if __name__ == "__main__": main()
